---
title: "Detect Image"
api: "POST /v1/detect"
description: "Detect AI-generated content in an image"
---

## Request

<ParamField body="image" type="string" required>
  Base64-encoded image data. Supports JPEG, PNG, WebP, and GIF formats.
</ParamField>

<ParamField body="models" type="string[]">
  Specific models to use for detection. Defaults to all available models.
</ParamField>

<ParamField body="threshold" type="number" default="0.5">
  Classification threshold (0-1). Images with confidence >= threshold are classified as AI-generated.
</ParamField>

## Response

<ResponseField name="prediction" type="string">
  Classification label: `"ai"`, `"real"`, or `"uncertain"`.
</ResponseField>

<ResponseField name="confidence" type="number">
  Overall confidence score (0-1).
</ResponseField>

<ResponseField name="ensemble_score" type="number">
  Weighted ensemble score (0-1).
</ResponseField>

<ResponseField name="verdict" type="string">
  Human-readable verdict: `"ai_generated"`, `"uncertain"`, or `"likely_real"`.
</ResponseField>

<ResponseField name="models_succeeded" type="number">
  Number of models that returned results successfully.
</ResponseField>

<ResponseField name="models_total" type="number">
  Total number of models invoked.
</ResponseField>

<ResponseField name="model_results" type="object">
  Per-model results keyed by model name.
  <Expandable title="ModelResult properties">
    <ResponseField name="score" type="number">
      Model confidence score (0-1).
    </ResponseField>
    <ResponseField name="status" type="string">
      Model status: `"success"` or `"error"`.
    </ResponseField>
    <ResponseField name="latency_ms" type="number">
      Inference latency in milliseconds.
    </ResponseField>
    <ResponseField name="error" type="string">
      Error message (if model failed).
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="detection_latency_ms" type="number">
  Total detection latency in milliseconds.
</ResponseField>

<ResponseField name="isFake" type="boolean | null">
  `true` if AI-generated, `false` if real, `null` if uncertain.
</ResponseField>

<ResponseField name="processingTimeMs" type="number">
  Total processing time in milliseconds.
</ResponseField>

<ResponseField name="imageHash" type="string">
  SHA-256 hash of the image.
</ResponseField>

<ResponseField name="cached" type="boolean">
  `true` if result was returned from cache.
</ResponseField>

<ResponseField name="timestamp" type="string">
  ISO 8601 timestamp of detection.
</ResponseField>

```python Python
from veri import VeriClient, DetectionOptions

client = VeriClient(api_key="your-api-key")

with open("image.jpg", "rb") as f:
    result = client.detect(
        f.read(),
        options=DetectionOptions(threshold=0.5)
    )

print(f"Prediction: {result.prediction}")
print(f"Is fake: {result.is_fake}")
print(f"Confidence: {result.confidence}")
```

```typescript TypeScript
import { VeriClient } from '@veri/sdk';

const client = new VeriClient({ apiKey: 'your-api-key' });

const result = await client.detect(imageBuffer, {
  threshold: 0.5
});

console.log(`Prediction: ${result.prediction}`);
console.log(`Is fake: ${result.isFake}`);
console.log(`Confidence: ${result.confidence}`);
```
